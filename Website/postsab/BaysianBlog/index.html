from scipy.special import factorial as f_
from scipy.special import comb
from scipy.stats import binom

def ProbIntegralValue(a,b):
    return f_(a)*f_(b)/f_(a+b+1)

def BayesianProbability(q,m,n,k):
    
    a = q
    b = m-a
    den = ProbIntegralValue(a,b)
    
    a = k+q
    b = n+m-a
    num = comb(n,k)*ProbIntegralValue(a,b)
    
    return num/den

def BayesianProbability2(q,m,n,k):
    p0 = BayesianProbability(q,m,n,k)
    
    a1 = n-k + m-q
    a2 = n-k -q  ## corr
    
    corr = a2/a1
    
    return p0*corr

def FreqProbability(q,m,n,k):
    
    p_hat = q/m
    prob = binom.pmf(k,n,p_hat)
    return prob

def ProbabilitySum(func, q,m,n,k):
    prob_v = np.array([func(q,m,n,i) for i in range(k,n+1)])
    #print(prob_v)
    return prob_v.sum()

def ProbabilitySumVector(func, q,m,n):
    def funcsum(k): 
        return ProbabilitySum(func, q,m,n,k)
    prob_v = np.array([funcsum(i) for i in range(0,n+1)])
    return prob_v
